  1. \displaystyle\frac{x^2-9}{6x+24} \times \displaystyle\frac{10x+40}{x(x+3)}
  2. \displaystyle\frac{x-5}{2x} \times \displaystyle\frac{4x^2}{2x-10}
  3. \displaystyle\frac{m-8}{m^2} \times 2m^2
  4. \displaystyle\frac{6x^2}{15} \div \displaystyle\frac{12x}{30x^3}
  5. \displaystyle\frac{3b+9}{b+3} \div \displaystyle\frac{b}{b}
  6. \displaystyle\frac{2y^3-4y^2}{3y^2-9y} \div \displaystyle\frac{y^2-4}{(y+2)(y-3)}
